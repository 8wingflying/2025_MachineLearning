import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler

# 1. Time Feature
df['hour'] = (df['Time'] / 3600) % 24
df['hour_sin'] = np.sin(2 * np.pi * df['hour'] / 24)
df['hour_cos'] = np.cos(2 * np.pi * df['hour'] / 24)

# 2. Amount Transform
df['LogAmount'] = np.log1p(df['Amount'])

# 3. Standardization
scaler = StandardScaler()
df[['Amount_scaled', 'LogAmount_scaled']] = scaler.fit_transform(df[['Amount', 'LogAmount']])

# 4. Amount Outlier Label
df['amount_z'] = (df['Amount'] - df['Amount'].mean()) / df['Amount'].std()
df['amount_outlier'] = (df['amount_z'] > 3).astype(int)

# 5. Time Segment
def time_segment(h):
    if 0 <= h < 6: return 'midnight'
    if 6 <= h < 12: return 'morning'
    if 12 <= h < 18: return 'afternoon'
    return 'evening'

df['time_segment'] = df['hour'].apply(time_segment)
df = pd.get_dummies(df, columns=['time_segment'])
